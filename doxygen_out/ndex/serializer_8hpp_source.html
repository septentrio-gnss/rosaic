96 public:
97 Reader(const uint8_t *data, uint32_t count,
98  const Options &options = Options()) :

101 typedef const uint8_t *iterator;
102 
103 iterator search()
104 {
105  if (found_) next();
106 
107  // Search for a message header
108  for( ; count_ > 0; --count_, ++data_) {
109  if (data_[0] == options_.sync_a &&
110  (count_ == 1 || data_[1] == options_.sync_b))
111  break;
112  }
113 
114  return data_;
115 }
116 
117 bool found()
118 {
119  if (found_) return true;
120  // Verify message is long enough to have sync chars, id, length & checksum
121  if (count_ < options_.wrapper_length()) return false;
122  // Verify the header bits
123  if (data_[0] != options_.sync_a || data_[1] != options_.sync_b)
124  return false;
125  // Verify that the buffer length is long enough based on the received
126  // message length
127  if (count_ < length() + options_.wrapper_length()) return false;
128 
129  found_ = true;
130  return true;
131 }
132 
133 iterator next() {
134  if (found()) {
135  uint32_t size = length() + options_.wrapper_length();
136  data_ += size; count_ -= size;
137  }
138  found_ = false;
139  return data_;
140 }
141 
142 iterator pos() {
143  return data_;
144 }
145 
146 iterator end() {
147  return data_ + count_;
148 }

150 uint8_t classId() { return data_[2]; }
151 uint8_t messageId() { return data_[3]; }
152 
153 uint32_t length() { return (data_[5] << 8) + data_[4]; }
154 const uint8_t *data() { return data_ + options_.header_length; }

156 uint16_t checksum() {
157  return *reinterpret_cast<const uint16_t *>(data_ + options_.header_length +
158  length());
159 }

161 template <typename T>
162 bool read(typename boost::call_traits<T>::reference message,
163  bool search = false) {
164  if (search) this->search();
165  if (!found()) return false;
166  if (!Message<T>::canDecode(classId(), messageId())) return false;
167 
168  uint16_t chk;
169  if (calculateChecksum(data_ + 2, length() + 4, chk) != this->checksum()) {
170  // checksum error
171  ROS_DEBUG("U-Blox read checksum error: 0x%02x / 0x%02x", classId(),
172  messageId());
173  return false;
174  }
175 
176  Serializer<T>::read(data_ + options_.header_length, length(), message);
177  return true;
178 }

184 }

205 template <typename T> bool write(const T& message,
206  uint8_t class_id = T::CLASS_ID,
207  uint8_t message_id = T::MESSAGE_ID) {
208  // Check for buffer overflow
209  uint32_t length = Serializer<T>::serializedLength(message);
210  if (size_ < length + options_.wrapper_length()) {
211  ROS_ERROR("u-blox write buffer overflow. Message %u / %u not written",
212  class_id, message_id);
213  return false;
214  }
215  // Encode the message and add it to the buffer
216  Serializer<T>::write(data_ + options_.header_length,
217  size_ - options_.header_length, message);
218  return write(0, length, class_id, message_id);
219 }
220 
221 bool write(const uint8_t* message, uint32_t length, uint8_t class_id,
222  uint8_t message_id) {
223  if (size_ < length + options_.wrapper_length()) {
224  ROS_ERROR("u-blox write buffer overflow. Message %u / %u not written",
225  class_id, message_id);
226  return false;
227  }
228  iterator start = data_;
229 
230  // write header
231  *data_++ = options_.sync_a;
232  *data_++ = options_.sync_b;
233  *data_++ = class_id;
234  *data_++ = message_id;
235  *data_++ = length & 0xFF;
236  *data_++ = (length >> 8) & 0xFF;
237  size_ -= options_.header_length;
238 
239  // write message
240  if (message) std::copy(message, message + length, data_);
241  data_ += length;
242  size_ -= length;
243 
244  // write checksum
245  uint8_t ck_a, ck_b;
246  calculateChecksum(start + 2, length + 4, ck_a, ck_b);
247  *data_++ = ck_a;
248  *data_++ = ck_b;
249  size_ -= options_.checksum_length;
250 
251  return true;
252 }
253 
254 iterator end() {
255  return data_;
256 }

259 iterator data_;
260 
261 uint32_t size_;
262 Options options_;
