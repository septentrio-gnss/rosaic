223  double UTCDoubleToSeconds(double utc_double)
224  {
225  uint32_t hours = static_cast<uint32_t>(utc_double) / 10000;
226  uint32_t minutes = (static_cast<uint32_t>(utc_double) - hours * 10000) / 100;
227  double seconds = utc_double - static_cast<double>(hours * 10000 + minutes * 100);
228  seconds += static_cast<double> (hours * 3600 + minutes * 60);
229  return seconds;
230  }

250  time_t UTCtoUnix(double utc_double)
251  {
252  time_t time_now = time(0);
253  struct tm * timeinfo;
254 
255  // The function localtime uses the value pointed by its argument to fill a tm structure with the values that represent the corresponding time, expressed for the local timezone.
256  timeinfo = localtime ( &time_now );
257 
258  uint32_t year = timeinfo->tm_year; // year, starting from 1900
259  uint32_t month = timeinfo->tm_mon; // months since January - [0,11]
260  uint32_t day = timeinfo->tm_mday; //day of the month - [1,31]
261  uint32_t hours = static_cast<uint32_t>(utc_double) / 10000;
262  uint32_t minutes = (static_cast<uint32_t>(utc_double) - hours * 10000) / 100;
263  uint32_t seconds = (static_cast<uint32_t>(utc_double) - hours * 10000 - minutes * 100);
264  if ((((int32_t) timeinfo->tm_hour) - ((int32_t) hours )) > 12) // E.g. in the UTC+14:00 time zone..
265  {
266  day -= 1;
267  }
268  if ((((int32_t) hours )- ((int32_t) timeinfo->tm_hour)) > 12)
269  {
270  day += 1;
271  }
272 
273  //ROS_DEBUG("Checking year %u, month %u, day %u", year, month, day);
274 
275  // Overwriting timeinfo with UTC time as extracted from utc_double..
276  timeinfo->tm_hour = hours; // hours since midnight - [0,23]
277  timeinfo->tm_min = minutes; // minutes after the hour - [0,59]
278  timeinfo->tm_sec = seconds; // seconds after the minute - [0,59]
279 
280  // Inverse of gmtime, the latter converts time_t (Unix time) to tm (UTC time)
281  time_t date = timegm ( timeinfo );
282 
283  //ROS_DEBUG("Until the given date, since 1970/01/01 %jd seconds have passed.\n", (intmax_t) date);
284  return date;
285  }
