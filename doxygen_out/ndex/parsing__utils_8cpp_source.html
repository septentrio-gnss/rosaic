117  double ParseDouble(const uint8_t* buffer)
118  {
119  double x;
120  std::copy(buffer, buffer + sizeof(double), reinterpret_cast<uint8_t*>(&x));
121  // Copies the elements in the range [buffer,buffer + sizeof(double)) into the range beginning at reinterpret_cast<uint8_t*>(&x).
122  // data_type *var_name = reinterpret_cast <data_type *>(pointer_variable) converts the pointer type, no return type
123  return x;

150  bool ParseInt16(const std::string& string, int16_t& value, int32_t base)
151  {
152  value = 0;
153  if (string.empty())
154  {
155  return true;
156  }
157 
158  int32_t tmp;
159  if (string_utilities::ToInt32(string, tmp, base) &&
160  tmp <= std::numeric_limits<int16_t>::max() &&
161  tmp >= std::numeric_limits<int16_t>::min())
162  {
163  value = static_cast<int16_t>(tmp);
164  return true;
165  }
166 
167  return false;
168  }

237  double UtcFloatToSeconds(double utc_float)
238  {
239  uint32_t hours = static_cast<uint32_t>(utc_float) / 10000;
240  uint32_t minutes = (static_cast<uint32_t>(utc_float) - hours * 10000) / 100;
241  double seconds = utc_float -
242  static_cast<double>(hours * 10000 + minutes * 100);
243  seconds += static_cast<double> (hours * 3600 + minutes * 60);
244  return seconds;
245  }

247  double ConvertDmsToDegrees(double dms)
248  {
249  uint32_t whole_degrees = static_cast<uint32_t>(dms) / 100;
250  double minutes = dms - static_cast<double>(whole_degrees * 100);
251  // 60 minutes in a degree
252  double degrees = static_cast<double>(whole_degrees) + minutes / 60.0;
253  return degrees;
254  }
