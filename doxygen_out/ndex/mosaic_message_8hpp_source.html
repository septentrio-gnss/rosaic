204  template <typename T>
205  bool mosaicMessage::read(typename boost::call_traits<T>::reference message, bool search)
206  {
207  //ROS_DEBUG("Inside mosaicMessage's read");
208  if (search) this->search();
209  if (!found()) return false;
210  if (data_[1] == SEP_SYNC_BYTE_2)
211  {
212  // CRC Check
213  bool CRCcheck = CRCIsValid(data_);
214  if (!CRCcheck)
215  {
216  ROS_DEBUG("CRC Check returned False for SBF block %s. Not a valid data block, perhaps noisy. Move on to next message/block of buffer..", this->MessageID().c_str());
217  return false;
218  }
219  uint32_t SBF_ID;
220  string_utilities::ToUInt32(this->MessageID(), SBF_ID);
221  switch(SBF_ID)
222  {
223  // Postion and velocity in XYZ
224  case 4006:
225  memcpy(&message, data_+8, sizeof(message)); // you cut out the block you are interested in via sizeof(..) since you do not care about the header of the block..
226  // read_timestamp... Caution: This should be part of Septentrio::readSerialPort(), since you want the timestamp when message came in, not after all these calculations...
227  // now do message.GPS_ms for instance
228  break;
229 
230  // Position Covariance block
231  case 5905:
232  memcpy(&message, data_+8, sizeof(message));
233  break;
234  // many more to be implemented...
235  }
236  }
237  else
238  {
239  boost::char_separator<char> sep("*");
240  typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
241  std::size_t end_point = std::min(static_cast<std::size_t>(this->end() - data_), static_cast<std::size_t>(82));
242  std::string block_in_string(reinterpret_cast<const char*>(data_), end_point); // We cannot uses overloaded '=' operator from string class since that converts char* to string
243  // Instead, reinterpret_cast<char*>(name) casts from uint8_t* to char* in an unsafe way but that's the one which should be used here. Then you call the ordinary constructor of std::string
244  tokenizer tokens(block_in_string, sep);
245  ROS_DEBUG("NMEA message is %s", (*tokens.begin()).c_str());
246  // typeid (explointing knowledge about T) instead of switch with message ID would not quite work (e.g. gcc gives you mangled names for instance), and even boost's fancy names are not that fancy..
247  switch(StringValues[this->MessageID()])
248  {
249  case evGPGGA:
250  std::string id = this->MessageID();
251  std::string pure_block = *tokens.begin();
252  boost::char_separator<char> sep_2(",", "", boost::keep_empty_tokens);
253  tokenizer tokens_2(pure_block, sep_2);
254  std::vector<std::string> body;
255  for (tokenizer::iterator tok_iter = tokens_2.begin(); tok_iter != tokens_2.end(); ++tok_iter) // perhaps str.erase from <string.h> would be faster, but i would not know what exactly to do..
256  {
257  // create NmeaSentence struct to pass to GpggaParser::ParseAscii
258  body.push_back(*tok_iter);
259  }
260  rosaic_driver::NMEASentence gga_message(id, body);
261  //ROS_DEBUG("First entry of vector is %s", body[0].c_str());
262  // Pass NmeaSentence struct to GGA-Parser
263  nmea_msgs::GpggaPtr gpgga_ros_message_ptr;
264  rosaic_driver::GpggaParser parser_obj;
265  try
266  {
267  gpgga_ros_message_ptr = parser_obj.ParseASCII(gga_message);
268  }
269  catch (rosaic_driver::ParseException& e)
270  {
271  std::cout << "GGA parsing failed: " << e.what() << "\n";
272  }
273  message = *gpgga_ros_message_ptr;
274  return true;
275  }
276  }
277  return true;
278  }
