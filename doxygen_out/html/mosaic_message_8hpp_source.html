275  template <typename T>
276  bool mosaicMessage::Read(typename boost::call_traits<T>::reference message, std::string message_key, bool search)
277  {
278  //ROS_DEBUG("message key is %s", message_key.c_str());
279  if (search) this->Search();
280  if (!Found()) return false;
281  if (this->IsSBF())
282  {
283  // If the CRC check is unsuccessful, throw an error message.
284  CRCcheck_ = CRCIsValid(data_);
285  if (!CRCcheck_)
286  {
287  throw std::runtime_error("CRC Check returned False. Not a valid data block, perhaps noisy. Ignore..");
288  }
289  }
290  switch(StringValues[message_key])
291  {
292  case evPVTCartesian: // Position and velocity in XYZ
293  { // The curly bracket here is crucial: Declarations inside a block remain inside, and will die at the end of the block. Otherwise variable overloading etc.
294  rosaic::PVTCartesianPtr msg = boost::make_shared<rosaic::PVTCartesian>();
295  PVTCartesian pvtcartesian;
296  memcpy(&pvtcartesian, data_, sizeof(pvtcartesian));
297  msg = PVTCartesianCallback(pvtcartesian);
298  msg->ROS_Header.seq = read_count;
299  msg->ROS_Header.frame_id = frame_id;
300  uint32_t TOW = *(reinterpret_cast<const uint32_t *>(data_ + 8));
301  ros::Time time_obj;
302  time_obj = TimestampSBF(TOW, use_GNSS_time);
303  msg->ROS_Header.stamp.sec = time_obj.sec;
304  msg->ROS_Header.stamp.nsec = time_obj.nsec;
305  msg->Block_Header.ID = 4006;
306  memcpy(&message, msg.get(), sizeof(*msg));
307  ++read_count;
308  break;
309  }
310  case evPVTGeodetic: // Position and velocity in geodetic coordinate frame (ENU frame)
311  {
312  rosaic::PVTGeodeticPtr msg = boost::make_shared<rosaic::PVTGeodetic>();
313  memcpy(&last_pvtgeodetic_, data_, sizeof(last_pvtgeodetic_));
314  msg = PVTGeodeticCallback(last_pvtgeodetic_);
315  msg->ROS_Header.seq = read_count;
316  msg->ROS_Header.frame_id = frame_id;
317  uint32_t TOW = *(reinterpret_cast<const uint32_t *>(data_ + 8));
318  ros::Time time_obj;
319  time_obj = TimestampSBF(TOW, use_GNSS_time);
320  msg->ROS_Header.stamp.sec = time_obj.sec;
321  msg->ROS_Header.stamp.nsec = time_obj.nsec;
322  msg->Block_Header.ID = 4007;
323  memcpy(&message, msg.get(), sizeof(*msg));
324  ++read_count;
325  break;
326  }
327 
328  case evPosCovGeodetic:
329  {
330  rosaic::PosCovGeodeticPtr msg = boost::make_shared<rosaic::PosCovGeodetic>();
331  memcpy(&last_poscovgeodetic_, data_, sizeof(last_poscovgeodetic_));
332  msg = PosCovGeodeticCallback(last_poscovgeodetic_);
333  msg->ROS_Header.seq = read_count;
334  msg->ROS_Header.frame_id = frame_id;
335  uint32_t TOW = *(reinterpret_cast<const uint32_t *>(data_ + 8));
336  ros::Time time_obj;
337  time_obj = TimestampSBF(TOW, use_GNSS_time);
338  msg->ROS_Header.stamp.sec = time_obj.sec;
339  msg->ROS_Header.stamp.nsec = time_obj.nsec;
340  msg->Block_Header.ID = 5906;
341  memcpy(&message, msg.get(), sizeof(*msg));
342  ++read_count;
343  break;
344  }
345  case evGPGGA:
346  {
347  boost::char_separator<char> sep("*");
348  typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
349  std::size_t end_point = std::min(static_cast<std::size_t>(this->End() - data_), static_cast<std::size_t>(82));
350  std::string block_in_string(reinterpret_cast<const char*>(data_), end_point);
351  tokenizer tokens(block_in_string, sep);
352 
353  std::string id = this->MessageID();
354  std::string one_message = *tokens.begin();
355  boost::char_separator<char> sep_2(",", "", boost::keep_empty_tokens);
356  tokenizer tokens_2(one_message, sep_2);
357  std::vector<std::string> body;
358  for (tokenizer::iterator tok_iter = tokens_2.begin(); tok_iter != tokens_2.end(); ++tok_iter) // perhaps str.erase from <string.h> would be faster, but i would not know what exactly to do..
359  {
360  body.push_back(*tok_iter);
361  }
362  // Create NmeaSentence struct to pass to GpggaParser::ParseASCII
363  rosaic_driver::NMEASentence gga_message(id, body);
364  rosaic::GpggaPtr gpgga_ros_message_ptr;
365  rosaic_driver::GpggaParser parser_obj;
366  try
367  {
368  gpgga_ros_message_ptr = parser_obj.ParseASCII(gga_message);
369  }
370  catch (rosaic_driver::ParseException& e)
371  {
372  throw std::runtime_error(e.what());
373  }
374  //ROS_DEBUG("ID is %s, size of message is %lu and sizeof ptr is %lu", gpgga_ros_message_ptr->message_id.c_str(), sizeof(message), sizeof(*gpgga_ros_message_ptr));
375  memcpy(&message, gpgga_ros_message_ptr.get(), sizeof(*gpgga_ros_message_ptr));
376  break;
377  }
378  case evNavSatFix:
379  {
380  sensor_msgs::NavSatFixPtr msg = boost::make_shared<sensor_msgs::NavSatFix>();
381  try
382  {
383  msg = NavSatFixCallback();
384  }
385  catch (std::runtime_error& e)
386  {
387  throw std::runtime_error(e.what());
388  }
389  msg->header.seq = read_count;
390  msg->header.frame_id = frame_id;
391  uint32_t TOW = last_pvtgeodetic_.TOW;
392  ros::Time time_obj;
393  time_obj = TimestampSBF(TOW, use_GNSS_time);
394  msg->header.stamp.sec = time_obj.sec;
395  msg->header.stamp.nsec = time_obj.nsec;
396  memcpy(&message, msg.get(), sizeof(*msg));
397  ++read_count;
398  break;
399  }
400  // Many more to be implemented...
401  }
402  return true;
403  }
404 }
