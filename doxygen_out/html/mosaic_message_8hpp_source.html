237  bool mosaicMessage::read(typename boost::call_traits<T>::reference message, bool search)
238  {
239  //ROS_DEBUG("Inside mosaicMessage's read");
240  if (search) this->search();
241  if (!found()) return false;
242  if (data_[0] == SEP_SYNC_BYTE_1 && data_[1] == SEP_SYNC_BYTE_2)
243  {
244  // CRC Check
245  bool CRCcheck = CRCIsValid(data_);
246  if (!CRCcheck)
247  {
248  std::stringstream ss;
249  ss << "CRC Check returned False. Not a valid data block, perhaps noisy. Ignore..";
250  throw std::runtime_error(ss.str().c_str());
251  }
252  uint32_t SBF_ID;
253  string_utilities::ToUInt32(this->MessageID(), SBF_ID);
254  switch(SBF_ID)
255  {
256  // Position and velocity in XYZ
257  case 4006:
258  { // The curly bracket here is crucial: declarations inside a block remain inside, and will die at the end of the block
259  rosaic::PVTCartesianPtr msg = boost::make_shared<rosaic::PVTCartesian>();
260  PVTCartesian pvt_cartesian;
261  memcpy(&pvt_cartesian, data_, sizeof(pvt_cartesian));
262  msg = PVTCartesianCallback(pvt_cartesian);
263  msg->ROS_Header.seq = read_count;
264  msg->ROS_Header.frame_id = frame_id;
265  uint32_t TOW = *(reinterpret_cast<const uint32_t *>(data_ + 8));
266  ros::Time time_obj;
267  time_obj = timestampSBF(TOW, use_GNSS_time);
268  msg->ROS_Header.stamp.sec = time_obj.sec;
269  msg->ROS_Header.stamp.nsec = time_obj.nsec;
270  msg->Block_Header.ID = 4006;
271  memcpy(&message, msg.get(), sizeof(*msg));
272  ++read_count;
273  break;
274  }
275 
276  // Position and velocity in geodetic coordinate frame (ENU frame)
277  case 4007:
278  { // The curly bracket here is crucial: declarations inside a block remain inside, and will die at the end of the block
279  rosaic::PVTGeodeticPtr msg = boost::make_shared<rosaic::PVTGeodetic>();
280  PVTGeodetic pvt_geodetic;
281  memcpy(&pvt_geodetic, data_, sizeof(pvt_geodetic));
282  msg = PVTGeodeticCallback(pvt_geodetic);
283  msg->ROS_Header.seq = read_count;
284  msg->ROS_Header.frame_id = frame_id;
285  uint32_t TOW = *(reinterpret_cast<const uint32_t *>(data_ + 8));
286  ros::Time time_obj;
287  time_obj = timestampSBF(TOW, use_GNSS_time);
288  msg->ROS_Header.stamp.sec = time_obj.sec;
289  msg->ROS_Header.stamp.nsec = time_obj.nsec;
290  msg->Block_Header.ID = 4007;
291  memcpy(&message, msg.get(), sizeof(*msg));
292  ++read_count;
293  break;
294  }
295 
296  // Position Covariance block
297  case 5905:
298  {
299  memcpy(&message, data_+8, sizeof(message));
300  break;
301  }
302  // many more to be implemented...
303  }
304  }
305  if ((data_[0] == SEP_SYNC_BYTE_1 && data_[1] == SEP_SYNC_BYTE_3) || (data_[0] == SEP_SYNC_BYTE_1 && data_[1] == SEP_SYNC_BYTE_4))
306  {
307  boost::char_separator<char> sep("*");
308  typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
309  std::size_t end_point = std::min(static_cast<std::size_t>(this->end() - data_), static_cast<std::size_t>(82));
310  std::string block_in_string(reinterpret_cast<const char*>(data_), end_point);
311  tokenizer tokens(block_in_string, sep);
312  ROS_DEBUG("NMEA message is %s", (*tokens.begin()).c_str());
313  switch(StringValues[this->MessageID()])
314  {
315  case evGPGGA:
316  {
317  std::string id = this->MessageID();
318  std::string pure_block = *tokens.begin();
319  boost::char_separator<char> sep_2(",", "", boost::keep_empty_tokens);
320  tokenizer tokens_2(pure_block, sep_2);
321  std::vector<std::string> body;
322  for (tokenizer::iterator tok_iter = tokens_2.begin(); tok_iter != tokens_2.end(); ++tok_iter) // perhaps str.erase from <string.h> would be faster, but i would not know what exactly to do..
323  {
324  body.push_back(*tok_iter);
325  }
326  // Create NmeaSentence struct to pass to GpggaParser::ParseASCII
327  rosaic_driver::NMEASentence gga_message(id, body);
328  nmea_msgs::GpggaPtr gpgga_ros_message_ptr;
329  rosaic_driver::GpggaParser parser_obj;
330  try
331  {
332  gpgga_ros_message_ptr = parser_obj.ParseASCII(gga_message);
333  }
334  catch (rosaic_driver::ParseException& e)
335  {
336  ROS_INFO("GGA parsing failed: %s\n", e.what());
337  throw std::runtime_error(e.what());
338  }
339  memcpy(&message, gpgga_ros_message_ptr.get(), sizeof(*gpgga_ros_message_ptr));
340  return true;
341  }
342  }
343  }
344  return true;
345  }
