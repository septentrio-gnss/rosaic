402  template <typename T>
403  bool mosaicMessage::Read(typename boost::call_traits<T>::reference message, std::string message_key, bool search)
404  {
405  //ROS_DEBUG("message key is %s", message_key.c_str());
406  if (search) this->Search();
407  if (!Found()) return false;
408  if (this->IsSBF())
409  {
410  // If the CRC check is unsuccessful, throw an error message.
411  CRCcheck_ = CRCIsValid(data_);
412  if (!CRCcheck_)
413  {
414  throw std::runtime_error("CRC Check returned False. Not a valid data block, perhaps noisy. Ignore..");
415  }
416  }
417  switch(StringValues[message_key])
418  {
419  case evPVTCartesian: // Position and velocity in XYZ
420  { // The curly bracket here is crucial: Declarations inside a block remain inside, and will die at the end of the block. Otherwise variable overloading etc.
421  rosaic::PVTCartesianPtr msg = boost::make_shared<rosaic::PVTCartesian>();
422  PVTCartesian pvtcartesian;
423  memcpy(&pvtcartesian, data_, sizeof(pvtcartesian));
424  msg = PVTCartesianCallback(pvtcartesian);
425  msg->ROS_Header.seq = count_pvtcartesian_;
426  msg->ROS_Header.frame_id = frame_id;
427  uint32_t TOW = *(reinterpret_cast<const uint32_t *>(data_ + 8));
428  ros::Time time_obj;
429  time_obj = TimestampSBF(TOW, use_GNSS_time);
430  msg->ROS_Header.stamp.sec = time_obj.sec;
431  msg->ROS_Header.stamp.nsec = time_obj.nsec;
432  msg->Block_Header.ID = 4006;
433  memcpy(&message, msg.get(), sizeof(*msg));
434  ++count_pvtcartesian_;
435  break;
436  }
437  case evPVTGeodetic: // Position and velocity in geodetic coordinate frame (ENU frame)
438  {
439  rosaic::PVTGeodeticPtr msg = boost::make_shared<rosaic::PVTGeodetic>();
440  memcpy(&last_pvtgeodetic_, data_, sizeof(last_pvtgeodetic_));
441  msg = PVTGeodeticCallback(last_pvtgeodetic_);
442  msg->ROS_Header.seq = count_pvtgeodetic_;
443  msg->ROS_Header.frame_id = frame_id;
444  uint32_t TOW = *(reinterpret_cast<const uint32_t *>(data_ + 8));
445  ros::Time time_obj;
446  time_obj = TimestampSBF(TOW, use_GNSS_time);
447  msg->ROS_Header.stamp.sec = time_obj.sec;
448  msg->ROS_Header.stamp.nsec = time_obj.nsec;
449  msg->Block_Header.ID = 4007;
450  memcpy(&message, msg.get(), sizeof(*msg));
451  ++count_pvtgeodetic_;
452  break;
453  }
454 
455  case evPosCovGeodetic:
456  {
457  rosaic::PosCovGeodeticPtr msg = boost::make_shared<rosaic::PosCovGeodetic>();
458  memcpy(&last_poscovgeodetic_, data_, sizeof(last_poscovgeodetic_));
459  msg = PosCovGeodeticCallback(last_poscovgeodetic_);
460  msg->ROS_Header.seq = count_poscovgeodetic_;
461  msg->ROS_Header.frame_id = frame_id;
462  uint32_t TOW = *(reinterpret_cast<const uint32_t *>(data_ + 8));
463  ros::Time time_obj;
464  time_obj = TimestampSBF(TOW, use_GNSS_time);
465  msg->ROS_Header.stamp.sec = time_obj.sec;
466  msg->ROS_Header.stamp.nsec = time_obj.nsec;
467  msg->Block_Header.ID = 5906;
468  memcpy(&message, msg.get(), sizeof(*msg));
469  ++count_poscovgeodetic_;
470  break;
471  }
472  case evAttEuler:
473  {
474  rosaic::AttEulerPtr msg = boost::make_shared<rosaic::AttEuler>();
475  memcpy(&last_atteuler_, data_, sizeof(last_atteuler_));
476  msg = AttEulerCallback(last_atteuler_);
477  msg->ROS_Header.seq = count_atteuler_;
478  msg->ROS_Header.frame_id = frame_id;
479  uint32_t TOW = *(reinterpret_cast<const uint32_t *>(data_ + 8));
480  ros::Time time_obj;
481  time_obj = TimestampSBF(TOW, use_GNSS_time);
482  msg->ROS_Header.stamp.sec = time_obj.sec;
483  msg->ROS_Header.stamp.nsec = time_obj.nsec;
484  msg->Block_Header.ID = 5938;
485  memcpy(&message, msg.get(), sizeof(*msg));
486  ++count_atteuler_;
487  break;
488  }
489  case evAttCovEuler:
490  {
491  rosaic::AttCovEulerPtr msg = boost::make_shared<rosaic::AttCovEuler>();
492  memcpy(&last_attcoveuler_, data_, sizeof(last_attcoveuler_));
493  msg = AttCovEulerCallback(last_attcoveuler_);
494  msg->ROS_Header.seq = count_attcoveuler_;
495  msg->ROS_Header.frame_id = frame_id;
496  uint32_t TOW = *(reinterpret_cast<const uint32_t *>(data_ + 8));
497  ros::Time time_obj;
498  time_obj = TimestampSBF(TOW, use_GNSS_time);
499  msg->ROS_Header.stamp.sec = time_obj.sec;
500  msg->ROS_Header.stamp.nsec = time_obj.nsec;
501  msg->Block_Header.ID = 5939;
502  memcpy(&message, msg.get(), sizeof(*msg));
503  ++count_attcoveuler_;
504  break;
505  }
506  case evGPGGA:
507  {
508  boost::char_separator<char> sep("*");
509  typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
510  std::size_t end_point = std::min(static_cast<std::size_t>(this->End() - data_), static_cast<std::size_t>(82));
511  std::string block_in_string(reinterpret_cast<const char*>(data_), end_point);
512  tokenizer tokens(block_in_string, sep);
513 
514  std::string id = this->MessageID();
515  std::string one_message = *tokens.begin();
516  boost::char_separator<char> sep_2(",", "", boost::keep_empty_tokens);
517  tokenizer tokens_2(one_message, sep_2);
518  std::vector<std::string> body;
519  for (tokenizer::iterator tok_iter = tokens_2.begin(); tok_iter != tokens_2.end(); ++tok_iter) // perhaps str.erase from <string.h> would be faster, but i would not know what exactly to do..
520  {
521  body.push_back(*tok_iter);
522  }
523  // Create NmeaSentence struct to pass to GpggaParser::ParseASCII
524  rosaic_driver::NMEASentence gga_message(id, body);
525  rosaic::GpggaPtr gpgga_ros_message_ptr;
526  rosaic_driver::GpggaParser parser_obj;
527  try
528  {
529  gpgga_ros_message_ptr = parser_obj.ParseASCII(gga_message);
530  }
531  catch (rosaic_driver::ParseException& e)
532  {
533  throw std::runtime_error(e.what());
534  }
535  gpgga_ros_message_ptr->header.seq = count_gpgga_;
536  //ROS_DEBUG("ID is %s, size of message is %lu and sizeof ptr is %lu", gpgga_ros_message_ptr->message_id.c_str(), sizeof(message), sizeof(*gpgga_ros_message_ptr));
537  memcpy(&message, gpgga_ros_message_ptr.get(), sizeof(*gpgga_ros_message_ptr));
538  ++count_gpgga_;
539  break;
540  }
541  case evNavSatFix:
542  {
543  sensor_msgs::NavSatFixPtr msg = boost::make_shared<sensor_msgs::NavSatFix>();
544  try
545  {
546  msg = NavSatFixCallback();
547  }
548  catch (std::runtime_error& e)
549  {
550  throw std::runtime_error(e.what());
551  }
552  msg->header.seq = count_navsatfix_;
553  msg->header.frame_id = frame_id;
554  uint32_t TOW = last_pvtgeodetic_.TOW;
555  ros::Time time_obj;
556  time_obj = TimestampSBF(TOW, use_GNSS_time);
557  msg->header.stamp.sec = time_obj.sec;
558  msg->header.stamp.nsec = time_obj.nsec;
559  memcpy(&message, msg.get(), sizeof(*msg));
560  ++count_navsatfix_;
561  break;
562  }
563  case evGPSFix:
564  {
565  gps_common::GPSFixPtr msg = boost::make_shared<gps_common::GPSFix>();
566  try
567  {
568  msg = GPSFixCallback();
569  }
570  catch (std::runtime_error& e)
571  {
572  throw std::runtime_error(e.what());
573  }
574  msg->header.seq = count_gpsfix_;
575  msg->status.header.seq = count_gpsfix_;
576  msg->header.frame_id = frame_id;
577  msg->status.header.frame_id = frame_id;
578  uint32_t TOW = last_pvtgeodetic_.TOW;
579  ros::Time time_obj;
580  time_obj = TimestampSBF(TOW, use_GNSS_time);
581  msg->header.stamp.sec = time_obj.sec;
582  msg->status.header.stamp.sec = time_obj.sec;
583  msg->header.stamp.nsec = time_obj.nsec;
584  msg->status.header.stamp.nsec = time_obj.nsec;
585  memcpy(&message, msg.get(), sizeof(*msg));
586  ++count_gpsfix_;
587  break;
588  }
589  case evChannelStatus:
590  {
591  memcpy(&last_channelstatus_, data_, sizeof(last_channelstatus_));
592  break;
593  }
594  case evMeasEpoch:
595  {
596  memcpy(&last_measepoch_, data_, sizeof(last_measepoch_));
597  break;
598  }
599  case evDOP:
600  {
601  memcpy(&last_dop_, data_, sizeof(last_dop_));
602  break;
603  }
604  case evVelCovGeodetic:
605  {
606  memcpy(&last_velcovgeodetic_, data_, sizeof(last_velcovgeodetic_));
607  break;
608  }
609  // Many more to be implemented...
610  }
611  return true;
612  }
