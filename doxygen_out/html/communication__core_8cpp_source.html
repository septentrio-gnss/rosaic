48 bool io_comm_mosaic::Comm_IO::InitializeSerial(std::string port, uint32_t baudrate,
49  std::string flowcontrol)
50 {
51  //ROS_DEBUG("Current debug value after calling initializeserial() method is %u", io_comm_mosaic::debug);
52  ROS_DEBUG("mosaic: Calling InitializeSerial method..");
53  serial_port_ = port;
54  baudrate_ = baudrate;
55  boost::shared_ptr<boost::asio::io_service> io_service(
56  new boost::asio::io_service);
57  // The io_context, of which io_service is a typedef of,
58  // represents your program's link to the operating system's I/O services.
59  boost::shared_ptr<boost::asio::serial_port> serial(
60  new boost::asio::serial_port(*io_service));
61  // To perform I/O operations the program needs an I/O object, here "serial".
62 
63  // We attempt the opening of the serial port..
64  try {
65  serial->open(serial_port_);
66  }
67  catch (std::runtime_error& e)
68  {
69  // and return an error message in case it fails.
70  throw std::runtime_error("mosaic-X5: Could not open serial port :"
71  + serial_port_ + " " + e.what());
72  return false;
73  }
74 
75  ROS_INFO("mosaic-X5: Opened serial port %s", serial_port_.c_str());
76  ROS_DEBUG("Our boost version is %u.", BOOST_VERSION);
77  if(BOOST_VERSION < 106600) // e.g. for ROS melodic (i.e. Ubuntu 18.04) it is 106501, standing for 1.65.1
78  {
79  // Workaround to set some options for the port manually,
80  // cf. https://github.com/mavlink/mavros/pull/971/files
81  // This function native_handle() may be used to obtain
82  // the underlying representation of the serial port.
83  // Conversion from type native_handle_type to int is done implicitly.
84  int fd = serial->native_handle();
85  termios tio;
86  // Get terminal attribute, follows the syntax
87  // int tcgetattr(int fd, struct termios *termios_p);
88  tcgetattr(fd, &tio);
89 
90  // Setting hardware flow control settings..
91  if (flowcontrol == "RTS|CTS") {
92  tio.c_iflag &= ~(IXOFF | IXON);
93  tio.c_cflag |= CRTSCTS;
94  } else {
95  tio.c_iflag &= ~(IXOFF | IXON);
96  tio.c_cflag &= ~CRTSCTS;
97  }
98  // Setting serial port to "raw" mode to prevent EOF exit..
99  cfmakeraw(&tio);
100 
101  // Commit settings, syntax is
102  // int tcsetattr(int fd, int optional_actions,
103  // const struct termios *termios_p);
104  tcsetattr(fd, TCSANOW, &tio);
105  }
106 
107  // Set the I/O manager
108  if (manager_)
109  {
110  ROS_ERROR("You have called InitializeSerial twice! Start all anew or call ResetSerial..");
111  return false;
112  }
113  ROS_DEBUG("Creating new Async-Manager..");
114  setManager(boost::shared_ptr<Manager>(new AsyncManager<boost::asio::serial_port>(serial, io_service)));
115 
116  //ROS_DEBUG("Finished creating new Async-Manager, have not yet called its read_callback_, since that will only be populated by the readCallback method of the CallbackHandlers class momentarily..");
117 
118  // Setting the baudrate, incrementally..
119  ROS_DEBUG("Gradually increasing the baudrate to the desired value...");
120  boost::asio::serial_port_base::baud_rate current_baudrate;
121  ROS_DEBUG("Initiated current_baudrate object...");
122  try
123  {
124  serial->get_option(current_baudrate); // Embed with throw to catch bad file descriptor error or similar ones..
125  // Often sets current_baudrate.value() magically to 115200
126  } catch(boost::system::system_error& e)
127  {
128 
129  ROS_ERROR("get_option failed due to %s", e.what());
130  ROS_INFO("Additional info about error is %s", boost::diagnostic_information(e).c_str());
131  /*
132  boost::system::error_code e_loop;
133  do // Caution: Might cause infinite loop..
134  {
135  serial->get_option(current_baudrate, e_loop);
136  } while(e_loop);
137  */
138  return 1;
139  }
140  // Gradually increase the baudrate to the desired value
141  // The desired baudrate can be lower or larger than the
142  // current baudrate; the for loop takes care of both scenarios.
143  ROS_DEBUG("Current baudrate is %u", current_baudrate.value());
144  for (uint8_t i = 0; i < sizeof(Baudrates)/sizeof(Baudrates[0]); i++)
145  {
146  if (current_baudrate.value() == baudrate_)
147  break;
148  // Break if the desired baudrate has been reached
149  if(current_baudrate.value() > Baudrates[i] && baudrate_ > Baudrates[i])
150  continue;
151  // Increment until Baudrate[i] matches current_baudrate
152  serial->set_option(boost::asio::serial_port_base::baud_rate(Baudrates[i]));
153  boost::this_thread::sleep(boost::posix_time::milliseconds(SetBaudrateSleepMs));
154  serial->get_option(current_baudrate);
155  ROS_DEBUG("mosaic: Set ASIO baudrate to %u", current_baudrate.value());
156  }
157  ROS_INFO("mosaic: Set ASIO baudrate to %u, leaving InitializeSerial() method", current_baudrate.value());
158  return true;
159 }
