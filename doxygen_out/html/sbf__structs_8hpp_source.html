37 /* Before the advent of the CPMF platform, double data types were always
38  * 32-bit aligned, meaning that the struct were aligned to an address
39  * that was divisible by 4. On the CPMF, double data types are 64-bit
40  * aligned. The "packed, aligned(4)" attribute combination is necessary
41  * to enforce 32-bit alignment for double data types and to port the sbf
42  * encoding/decoding functionality to the CPMF.
43  */
44 // The aligned variable attribute specifies a minimum alignment for the variable or structure field, measured in bytes.
45 // The aligned attribute only increases the alignment for a struct or struct member. For a variable that is not in a structure,
46 // the minimum alignment is the natural alignment of the variable type. To set the alignment in a structure to any value greater than 0,
47 // use the packed variable attribute. Without packed, the minimum alignment is the natural alignment of the variable type.
48 
49 # define SBFDOUBLE double __attribute__((packed, aligned(4)))
50 #else
51 # define SBFDOUBLE double
52 #endif

450 struct GPSNAV
451 {
452  BlockHeader_t Header;
453 
454  /* Time Header */
455  uint32_t TOW;
456  uint16_t WNc;
457 
458  uint8_t PRN;
459  uint8_t Reserved;
460  uint16_t WN; /* Week number (modulo 1024). [061-070:1] */
461  uint8_t CAorPonL2; /* C/A code or P code on L2. [071-072:1] */
462  uint8_t URA; /* User range accuracy index. [073-076:1] */
463  uint8_t health; /* SV health. [077-082:1] */
464  uint8_t L2DataFlag; /* L2 P data flag. [091-091:1] */
465  uint16_t IODC; /* Issue of data clock. [083-084:1] [211-218:1] */
466  uint8_t IODE2; /* Issue of data eph. frame 2. [061-068:2] */
467  uint8_t IODE3; /* Issue of data eph. frame 3. [271-278:3] */
468  uint8_t FitIntFlg; /* fit interval flag [287-287:2] */
469  uint8_t dummy; /* introduced w.r.t. 32-bits memory alignment */
470  float T_gd; /* Correction term T_gd (s). [197-204:1] */
471  uint32_t t_oc; /* Clock correction t_oc (s). [219-234:1] */
472  float a_f2; /* Clock correction a_f2 (s/s^2). [241-248:1] */
473  float a_f1; /* Clock correction a_f1 (s/s). [249-264:1] */
474  float a_f0; /* Clock correction a_f0 (s). [271-292:1] */
475  float C_rs; /* radius sin ampl (m) [069-084:2] */
476  float DEL_N; /* mean motion diff (semi-circles/s) [091-106:2] */
477  SBFDOUBLE M_0; /* Mean Anom (semi-circles) [107-114:2] [121-144:2] */
478  float C_uc; /* lat cosine ampl (r) [151-166:2] */
479  SBFDOUBLE e; /* Eccentricity [167-174:2] [181-204:2] */
480  float C_us; /* Lat sine ampl (r) [211-226:2] */
481  SBFDOUBLE SQRT_A; /* SQRT(A) (m^1/2) [227-234:2 241-264:2] */
482  uint32_t t_oe; /* Reference time of ephemeris (s) [271-286:2] */
483  float C_ic; /* inclin cos ampl (r) [061-076:3] */
484  SBFDOUBLE OMEGA_0; /* Right Ascen at TOA (semi-circles) [077-084:3] [091-114:3] */
485  float C_is; /* inclin sin ampl (r) [121-136:3] */
486  SBFDOUBLE i_0; /* Orbital Inclination (semi-circles) [137-144:3] [151-174:3] */
487  float C_rc; /* radius cos ampl (m) [181-196:3] */
488  SBFDOUBLE omega; /* Argument of Perigee(semi-circle) [197-204:3] [211-234:3] */
489  float OMEGADOT; /* Rate of Right Ascen(semi-circles/s) [241-264:3] */
490  float IDOT; /* Rate of inclin (semi-circles/s) [279-292:3] */
491  uint16_t WNt_oc; /* modified WN to go with t_oc (still modulo 1024) */
492  uint16_t WNt_oe; /* modified WN to go with t_oe (still modulo 1024) */
493 };
