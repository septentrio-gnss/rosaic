38 /* Before the advent of the CPMF platform, double data types were always
39  * 32-bit aligned, meaning that the struct were aligned to an address
40  * that was divisible by 4. On the CPMF, double data types are 64-bit
41  * aligned. The "packed, aligned(4)" attribute combination is necessary
42  * to enforce 32-bit alignment for double data types and to port the sbf
43  * encoding/decoding functionality to the CPMF.
44  */
45 // The aligned variable attribute specifies a minimum alignment for the variable or structure field, measured in bytes.
46 // The aligned attribute only increases the alignment for a struct or struct member. For a variable that is not in a structure,
47 // the minimum alignment is the natural alignment of the variable type. To set the alignment in a structure to any value greater than 0,
48 // use the packed variable attribute. Without packed, the minimum alignment is the natural alignment of the variable type.
49 
50 # define SBFDOUBLE double __attribute__((packed, aligned(4)))
51 #else
52 # define SBFDOUBLE double
53 #endif

477 struct GPSNAV
478 {
479  BlockHeader_t Header;
480 
481  /* Time Header */
482  uint32_t TOW;
483  uint16_t WNc;
484 
485  uint8_t PRN;
486  uint8_t Reserved;
487  uint16_t WN; /* Week number (modulo 1024). [061-070:1] */
488  uint8_t CAorPonL2; /* C/A code or P code on L2. [071-072:1] */
489  uint8_t URA; /* User range accuracy index. [073-076:1] */
490  uint8_t health; /* SV health. [077-082:1] */
491  uint8_t L2DataFlag; /* L2 P data flag. [091-091:1] */
492  uint16_t IODC; /* Issue of data clock. [083-084:1] [211-218:1] */
493  uint8_t IODE2; /* Issue of data eph. frame 2. [061-068:2] */
494  uint8_t IODE3; /* Issue of data eph. frame 3. [271-278:3] */
495  uint8_t FitIntFlg; /* fit interval flag [287-287:2] */
496  uint8_t dummy; /* introduced w.r.t. 32-bits memory alignment */
497  float T_gd; /* Correction term T_gd (s). [197-204:1] */
498  uint32_t t_oc; /* Clock correction t_oc (s). [219-234:1] */
499  float a_f2; /* Clock correction a_f2 (s/s^2). [241-248:1] */
500  float a_f1; /* Clock correction a_f1 (s/s). [249-264:1] */
501  float a_f0; /* Clock correction a_f0 (s). [271-292:1] */
502  float C_rs; /* radius sin ampl (m) [069-084:2] */
503  float DEL_N; /* mean motion diff (semi-circles/s) [091-106:2] */
504  SBFDOUBLE M_0; /* Mean Anom (semi-circles) [107-114:2] [121-144:2] */
505  float C_uc; /* lat cosine ampl (r) [151-166:2] */
506  SBFDOUBLE e; /* Eccentricity [167-174:2] [181-204:2] */
507  float C_us; /* Lat sine ampl (r) [211-226:2] */
508  SBFDOUBLE SQRT_A; /* SQRT(A) (m^1/2) [227-234:2 241-264:2] */
509  uint32_t t_oe; /* Reference time of ephemeris (s) [271-286:2] */
510  float C_ic; /* inclin cos ampl (r) [061-076:3] */
511  SBFDOUBLE OMEGA_0; /* Right Ascen at TOA (semi-circles) [077-084:3] [091-114:3] */
512  float C_is; /* inclin sin ampl (r) [121-136:3] */
513  SBFDOUBLE i_0; /* Orbital Inclination (semi-circles) [137-144:3] [151-174:3] */
514  float C_rc; /* radius cos ampl (m) [181-196:3] */
515  SBFDOUBLE omega; /* Argument of Perigee(semi-circle) [197-204:3] [211-234:3] */
516  float OMEGADOT; /* Rate of Right Ascen(semi-circles/s) [241-264:3] */
517  float IDOT; /* Rate of inclin (semi-circles/s) [279-292:3] */
518  uint16_t WNt_oc; /* modified WN to go with t_oc (still modulo 1024) */
519  uint16_t WNt_oe; /* modified WN to go with t_oe (still modulo 1024) */
520 };
