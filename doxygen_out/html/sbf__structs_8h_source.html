372 struct GPSNAV
373 {
374  BlockHeader Header;
375 
376  /* Time Header */
377  uint32_t TOW;
378  uint16_t WNc;
379 
380  uint8_t PRN;
381  uint8_t Reserved;
382  uint16_t WN; /* Week number (modulo 1024). [061-070:1] */
383  uint8_t CAorPonL2; /* C/A code or P code on L2. [071-072:1] */
384  uint8_t URA; /* User range accuracy index. [073-076:1] */
385  uint8_t health; /* SV health. [077-082:1] */
386  uint8_t L2DataFlag; /* L2 P data flag. [091-091:1] */
387  uint16_t IODC; /* Issue of data clock. [083-084:1] [211-218:1] */
388  uint8_t IODE2; /* Issue of data eph. frame 2. [061-068:2] */
389  uint8_t IODE3; /* Issue of data eph. frame 3. [271-278:3] */
390  uint8_t FitIntFlg; /* fit interval flag [287-287:2] */
391  uint8_t dummy; /* introduced w.r.t. 32-bits memory alignment */
392  float T_gd; /* Correction term T_gd (s). [197-204:1] */
393  uint32_t t_oc; /* Clock correction t_oc (s). [219-234:1] */
394  float a_f2; /* Clock correction a_f2 (s/s^2). [241-248:1] */
395  float a_f1; /* Clock correction a_f1 (s/s). [249-264:1] */
396  float a_f0; /* Clock correction a_f0 (s). [271-292:1] */
397  float C_rs; /* radius sin ampl (m) [069-084:2] */
398  float DEL_N; /* mean motion diff (semi-circles/s) [091-106:2] */
399  SBFDOUBLE M_0; /* Mean Anom (semi-circles) [107-114:2] [121-144:2] */
400  float C_uc; /* lat cosine ampl (r) [151-166:2] */
401  SBFDOUBLE e; /* Eccentricity [167-174:2] [181-204:2] */
402  float C_us; /* Lat sine ampl (r) [211-226:2] */
403  SBFDOUBLE SQRT_A; /* SQRT(A) (m^1/2) [227-234:2 241-264:2] */
404  uint32_t t_oe; /* Reference time of ephemeris (s) [271-286:2] */
405  float C_ic; /* inclin cos ampl (r) [061-076:3] */
406  SBFDOUBLE OMEGA_0; /* Right Ascen at TOA (semi-circles) [077-084:3] [091-114:3] */
407  float C_is; /* inclin sin ampl (r) [121-136:3] */
408  SBFDOUBLE i_0; /* Orbital Inclination (semi-circles) [137-144:3] [151-174:3] */
409  float C_rc; /* radius cos ampl (m) [181-196:3] */
410  SBFDOUBLE omega; /* Argument of Perigee(semi-circle) [197-204:3] [211-234:3] */
411  float OMEGADOT; /* Rate of Right Ascen(semi-circles/s) [241-264:3] */
412  float IDOT; /* Rate of inclin (semi-circles/s) [279-292:3] */
413  uint16_t WNt_oc; /* modified WN to go with t_oc (still modulo 1024) */
414  uint16_t WNt_oe; /* modified WN to go with t_oe (still modulo 1024) */
415 };
