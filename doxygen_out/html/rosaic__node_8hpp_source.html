168  template <typename U>
169  bool GetROSInt(const std::string& key, U &u) {
170  int param;
171  if (!nh->getParam(key, param)) return false;
172  U min = std::numeric_limits<U>::lowest();
173  U max = std::numeric_limits<U>::max();
174  try
175  {
176  CheckRange((U) param, min, max, key);
177  }
178  catch (std::runtime_error& e)
179  {
180  std::ostringstream ss;
181  ss << e.what();
182  ROS_INFO("%s", ss.str().c_str());
183  }
184  u = (U) param;
185  return true;
186  }
