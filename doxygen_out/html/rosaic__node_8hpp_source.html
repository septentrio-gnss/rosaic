170  template <typename U>
171  bool GetROSInt(const std::string& key, U &u) {
172  int param;
173  if (!nh->getParam(key, param)) return false;
174  U min = std::numeric_limits<U>::lowest();
175  U max = std::numeric_limits<U>::max();
176  try
177  {
178  CheckRange((U) param, min, max, key);
179  }
180  catch (std::runtime_error& e)
181  {
182  std::ostringstream ss;
183  ss << e.what();
184  ROS_INFO("%s", ss.str().c_str());
185  }
186  u = (U) param;
187  return true;
188  }
