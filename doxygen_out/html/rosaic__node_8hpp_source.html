171  template <typename U>
172  bool GetROSInt(const std::string& key, U &u) {
173  int param;
174  if (!nh->getParam(key, param)) return false;
175  U min = std::numeric_limits<U>::lowest();
176  U max = std::numeric_limits<U>::max();
177  try
178  {
179  CheckRange((U) param, min, max, key);
180  }
181  catch (std::runtime_error& e)
182  {
183  std::ostringstream ss;
184  ss << e.what();
185  ROS_INFO("%s", ss.str().c_str());
186  }
187  u = (U) param;
188  return true;
189  }
