164  template <typename U>
165  bool GetROSInt(const std::string& key, U &u) {
166  int param;
167  if (!nh->getParam(key, param)) return false;
168  // Check the bounds
169  U min = std::numeric_limits<U>::lowest();
170  U max = std::numeric_limits<U>::max();
171  checkRange((U) param, min, max, key);
172  // set the output
173  u = (U) param;
174  return true;
175  }
