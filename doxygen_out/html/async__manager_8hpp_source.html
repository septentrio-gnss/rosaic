242  AsyncManager<StreamT>::AsyncManager(boost::shared_ptr<StreamT> stream,
243  boost::shared_ptr<boost::asio::io_service> io_service,
244  std::size_t buffer_size) : timer_(*(io_service.get()), boost::posix_time::seconds(1)), stopping_(false), buffer_size_(buffer_size), count_max_(6) // Since buffer_size = 8912 in declaration, no need in definition any more (even yields error message, since "overwrite").
245  {
246  ROS_DEBUG("Setting the stream private variable of the AsyncManager class.");
247  do_read_count_ = 0;
248  stream_ = stream;
249  io_service_ = io_service;
250  in_.resize(buffer_size_);
251 
252  out_.reserve(buffer_size_); // Note that std::vector::reserve() requests to reserve vector capacity be at least enough to contain n elements. It leaves the allocated memory uninitialized.
253  // Reallocation happens if there is need of even more space. Note that size() is not changed by this command.
254 
255  io_service_->post(boost::bind(&AsyncManager<StreamT>::DoRead, this));
256  // This function is used to ask the io_service to execute the given handler, but without allowing the io_service to call the handler from inside this function.
257  // The function signature of the handler must be: void handler();
258  // The io_service guarantees that the handler (given as parameter) will only be called in a thread in which the run(), run_one(), poll() or poll_
259  // one() member functions is currently being invoked.
260  // So the fundamental difference is that dispatch will execute the work right away if it can and queue it otherwise while post queues the work no matter what.
261  async_background_thread_.reset(new boost::thread(boost::bind(&boost::asio::io_service::run, io_service_))); // io_service_ is already pointer
262  // If the value of the pointer for the current thread is changed using reset(), then the previous value is destroyed by calling the cleanup routine.
263  // Alternatively, the stored value can be reset to NULL and the prior value returned by calling the release() member function,
264  // allowing the application to take back responsibility for destroying the object.
265  uint16_t count = 0;
266  boost::thread(boost::bind(&AsyncManager::CallAsyncWait, this, &count));
267  } // Calls std::terminate() on thread just created

293  template <typename StreamT>
294  void AsyncManager<StreamT>::AsyncReadSomeHandler(const boost::system::error_code& error,
295  std::size_t bytes_transfered)
296  {
297  if (error)
298  {
299  ROS_ERROR("mosaic ASIO input buffer read error: %s, %li",
300  error.message().c_str(), bytes_transfered); // The c_str() method is also part of the <string> header.
301  }
302  else if (bytes_transfered > 0)
303  {
304  if (read_callback_) //Will be false in InitializeSerial (first call)
305  {
306  ROS_DEBUG("Launching ReadCallback thread, with bytes_transferred being %u", (unsigned int) bytes_transfered);
307  std::vector<uint8_t> copied_buffer = in_;
308  // Launch new thread, detached right afterwards..
309  boost::thread temporary_thread(read_callback_, copied_buffer, bytes_transfered);
310  temporary_thread.detach();
311  std::vector<uint8_t> empty;
312  in_ = empty;
313  in_.resize(buffer_size_);
314  }
315  }
316 
317  if (!stopping_)
318  io_service_->post(boost::bind(&AsyncManager<StreamT>::DoRead, this));
319  }

334  template <typename StreamT>
335  void AsyncManager<StreamT>::Wait(uint16_t* count) //const boost::posix_time::time_duration& timeout)
336  {
337  if (*count < count_max_)
338  {
339  ++(*count);
340  timer_.expires_at(timer_.expires_at() + boost::posix_time::seconds(1));
341  if (!(*count == count_max_))
342  {
343  timer_.async_wait(boost::bind(&AsyncManager::Wait, this, count));
344  }
345  }
346  if ((*count == count_max_) && (do_read_count_ < 3)) // Why 3? Even if there are no incoming messages, doRead() is called once.
347  // It will be called a second time in TCP/IP mode since (just example) "IP10<" is transmitted.
348  {
349  ROS_INFO("No incoming messages, driver stopped, ros::spin() will spin forever unless you hit Ctrl+C.");
350  async_background_thread_->interrupt();
351  }
352  }
