125 std::string io_comm_mosaic::mosaicMessage::MessageID()
126 {
127  if (data_[1] == SEP_SYNC_BYTE_2)
128  {
129  // Construct bit mask:
130  uint16_t mask = 0;
131  uint16_t value = (*reinterpret_cast<const uint16_t *>(data_ + 4)) & (~mask << 3); // Bitwise AND gives us first 13 bits unchanged, rest set to zero
132  std::stringstream ss;
133  ss << value;
134  return ss.str();
135  }
136  else

141  std::string block_in_string(reinterpret_cast<const char*>(data_), end_point);
142  tokenizer tokens(block_in_string,sep);
