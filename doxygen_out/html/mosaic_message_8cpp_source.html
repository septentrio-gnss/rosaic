122 ros::Time io_comm_mosaic::timestampSBF(uint32_t TOW, bool use_GNSS)
123 {
124  if (use_GNSS)
125  {
126  uint16_t hh = (TOW%(1000*60*60*24))/(60*60*1000);
127  uint16_t mm = ((TOW%(1000*60*60*24))-hh*(60*60*1000))/(60*1000);
128  uint16_t ss = ((TOW%(1000*60*60*24))-hh*(60*60*1000)-mm*(60*1000))/(1000);
129  uint16_t hs = ((TOW%(1000*60*60*24))-hh*(60*60*1000)-mm*(60*1000)-ss*1000)/10; // hundredths of a second
130  if (ss >= 18)
131  {
132  ss = ss - 18;
133  }
134  else
135  {
136  if (mm >= 1)
137  {
138  --mm;
139  ss = 60 - (18 - ss);
140  }
141  else
142  {
143  if (hh >= 1)
144  {
145  --hh;
146  mm = 59;
147  ss = 60 - (18 - ss);
148  }
149  else
150  {
151  hh = 23;
152  mm = 59;
153  ss = 60 - (18 - ss);
154  }
155  }
156  }
157  boost::format fmt = boost::format("%02u%02u%02u.%02u") % hh % mm % ss % hs;
158  std::string utc_string = fmt.str();
159  //ROS_DEBUG("UTC string (still need to subtract 18 seconds) is %s", utc_string.c_str());
160  double utc_double;
161  string_utilities::ToDouble(utc_string, utc_double);
162  time_t unix_time_seconds = rosaic_driver::UTCtoUnix(utc_double);
163  uint32_t unix_time_nanoseconds = (static_cast<uint32_t>(utc_double*100)%100)*10000000; // works since there are two digits after the decimal point in the utc_double
164  ros::Time time_obj(unix_time_seconds, unix_time_nanoseconds);
165  return time_obj;
166  }
167  else
168  {
169  time_t time_now = time(NULL);
170  ros::Time time_obj((uint32_t) time_now, 0);
171  return time_obj;
172  }
173 }

265 std::string io_comm_mosaic::mosaicMessage::MessageID()
266 {
267  if (data_[0] == SEP_SYNC_BYTE_1 && data_[1] == SEP_SYNC_BYTE_2)
268  {
269  //ROS_DEBUG("MessageID if clause");
270  // Construct bit mask:
271  uint16_t mask = 8191; // It is not as stated in the firmware: !first! three bits are for revision (not last 3), and rest for block number
272  uint16_t value = (*(reinterpret_cast<const uint16_t*>(data_+4))) & mask; // Bitwise AND gives us first 3 bits set to zero, rest unchanged
273  std::stringstream ss;
274  ss << value;
275  return ss.str();
276  }
277  if ((data_[0] == SEP_SYNC_BYTE_1 && data_[1] == SEP_SYNC_BYTE_3) || (data_[0] == SEP_SYNC_BYTE_1 && data_[1] == SEP_SYNC_BYTE_4))
278  {
279  boost::char_separator<char> sep(",");
280  typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
281  std::size_t end_point = std::min(static_cast<std::size_t>(this->end() - data_), static_cast<std::size_t>(82));
282  std::string block_in_string(reinterpret_cast<const char*>(data_), end_point);
283  tokenizer tokens(block_in_string,sep);
284  return *tokens.begin();
285  }
286  return std::string(); // less work than return "";
287 }
