136  void Handle(mosaicMessage& mMessage, std::string message_key)
137  {
138  boost::mutex::scoped_lock lock(mutex_);
139  try
140  {
141  if (!mMessage.Read<T>(message_, message_key))
142  {
143  std::ostringstream ss;
144  ss << "Read unsuccessful: mosaic decoder error for message with ID (empty field if non-determinable" << mMessage.MessageID() << ". ";
145  ROS_INFO_COND(debug >= 2, "%s", ss.str().c_str());
146  return;
147  }
148  } catch (std::runtime_error& e)
149  {
150  std::ostringstream ss;
151  ss << "Read unsuccessful: mosaic decoder error for message with ID " << mMessage.MessageID() << ".\nReason: " << e.what();
152  ROS_INFO_COND(debug >= 2, "%s", ss.str().c_str());
153  return;
154  }
155 
156  condition_.notify_all();
157  if (func_) func_(message_);
158  }
159 
160  private:
161  Callback func_;
162  T message_;
163  };

194  void Handle(mosaicMessage& mMessage)
195  {
196  // Find the ROS message callback handler for equivalent mosaic message at hand & call it
197  boost::mutex::scoped_lock lock(callback_mutex_);
198  CallbackMap::key_type key = mMessage.MessageID();
199  std::string ID_temp = mMessage.MessageID();
200  if (!(ID_temp == "4013" || ID_temp == "4027" || ID_temp == "4001"|| ID_temp == "5908")) // We only want to handle SatVisibility and ChannelStatus blocks in case GPSFix messages are to be published, see few lines below
201  {
202  for (CallbackMap::iterator callback = callbackmap_.lower_bound(key); callback != callbackmap_.upper_bound(key); ++callback)
203  {
204  callback->second->Handle(mMessage, callback->first);
205  }
206  }
207  // Call NavSatFix callback function if it was added
208  if (publish_navsatfix)
209  {
210  CallbackMap::key_type key = "NavSatFix";
211  std::string ID_temp = mMessage.MessageID();
212  if (ID_temp == "4007") // If no new PVTGeodetic block is coming in, there is no need to publish NavSatFix anew.
213  // It would be wasteful to also call the NavSatFix handle if the PosCovGeodetic block comes in.
214  {
215  for (CallbackMap::iterator callback = callbackmap_.lower_bound(key); callback != callbackmap_.upper_bound(key); ++callback)
216  {
217  callback->second->Handle(mMessage, callback->first);
218  }
219  }
220  }
221  if (publish_gpsfix)
222  {
223  CallbackMap::key_type key1 = "GPSFix";
224  std::string ID_temp = mMessage.MessageID();
225  if (ID_temp == "4007") // If no new PVTGeodetic block is coming in, there is no need to publish GPSFix anew.
226  // It would be wasteful to also call the GPSFix handle when any of the PosCovGeodetic, SatVisibility, ChannelStatus etc. blocks comes in.
227  {
228  for (CallbackMap::iterator callback = callbackmap_.lower_bound(key1); callback != callbackmap_.upper_bound(key1); ++callback)
229  {
230  callback->second->Handle(mMessage, callback->first);
231  }
232  }
233  CallbackMap::key_type key2 = mMessage.MessageID();
234  if (ID_temp == "4013" || ID_temp == "4027" || ID_temp == "4001" || ID_temp == "5908")
235  // Even though we are not interested in publishing SatVisibility (4012) and ChannelStatus (4013) ROS messages, we have to save
236  // some contents of these incoming blocks in order to publish the GPSFix message. When these 2 blocks come in, the unchanged GPSFix message is published anew.
237  {
238  for (CallbackMap::iterator callback = callbackmap_.lower_bound(key2); callback != callbackmap_.upper_bound(key2); ++callback)
239  {
240  callback->second->Handle(mMessage, callback->first);
241  }
242  }
243  }

256  template <typename T>
257  bool Poll(T& message, std::string message_key, const boost::posix_time::time_duration& timeout)
258  {
259  bool result = false;
260  // Insert callback handler to C++ map
261  callback_mutex_.lock();
262  CallbackHandler<T>* handler = new CallbackHandler<T>(); // This calls the constructor with an empty function.
263  CallbackMap::iterator callback = callbackmap_.insert(std::make_pair(message_key, boost::shared_ptr<AbstractCallbackHandler>(handler)));
264  callback_mutex_.unlock();
265 
266  // Wait for the message
267  if (handler->Wait(timeout))
268  {
269  message = handler->Get();
270  result = true;
271  }
272 
273  // Remove the callback handler
274  callback_mutex_.lock();
275  callbackmap_.erase(callback);
276  callback_mutex_.unlock();
277  return result;
278  }

285  void ReadCallback(std::vector<uint8_t> data, std::size_t& size)
286  {
287  uint8_t* data_raw = &data[0];
288  mosaicMessage mMessage(data_raw, size);
289  // Read !all! (there might be many) messages in the buffer
290  while (mMessage.Search() != mMessage.End() && mMessage.Found())
291  {
292  if (debug >= 3)
293  {
294  // Print the found message (if NMEA) or just show messageID (if SBF)..
295  if (mMessage.IsSBF())
296  {
297  unsigned long sbf_block_length;
298  sbf_block_length = (unsigned long) mMessage.BlockLength(); // C-like cast notation (since functional notation did not work, although https://www.cplusplus.com/doc/tutorial/typecasting/ suggests otherwise)
299  ROS_DEBUG("Driver reading SBF block %s with %lu bytes...", mMessage.MessageID().c_str(), sbf_block_length); // Recall: The long data type is at least 32 bits.
300  }
301  if (mMessage.IsNMEA())
302  {
303  std::ostringstream oss;
304  boost::char_separator<char> sep("\r");
305  typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
306  std::size_t nmea_size = mMessage.SegmentEnd();
307  std::string block_in_string(reinterpret_cast<const char*>(mMessage.Pos()), nmea_size); // Syntax: new_string_name (const char* s, size_t n); size_t is either 2 or 8 bytes, depending on your system
308  // NMEA 0183 messages are at most 82 characters long, , including the $ or ! starting character and the ending <LF>
309  tokenizer tokens(block_in_string, sep);
310  ROS_DEBUG("The NMEA message is made up of %li bytes and is ready to be iterated. It reads: %s", nmea_size, (*tokens.begin()).c_str());
311  }
312  if (mMessage.IsResponse())
313  {
314  std::ostringstream oss;
315  std::size_t response_size = mMessage.SegmentEnd();
316  std::string block_in_string(reinterpret_cast<const char*>(mMessage.Pos()), response_size);
317  ROS_DEBUG("mosaic's response contains %li bytes and reads:\n %s", response_size, block_in_string.c_str());
318  {
319  boost::mutex::scoped_lock lock(response_mutex);
320  response_received = true;
321  lock.unlock();
322  response_condition.notify_one();
323  }
324  if (mMessage.IsErrorMessage())
325  {
326  ROS_ERROR("Invalid command just sent to mosaic!");
327  }
328  }
329  }
330  //ROS_DEBUG("Handing over from readcallback to Handle while count is %d", mMessage.GetCount());
331  Handle(mMessage);
332  }
333  }
