212  template <typename T>
213  bool poll(T& message, std::string message_ID, const boost::posix_time::time_duration& timeout) {
214  bool result = false;
215  callback_mutex_.lock();
216  CallbackHandler_<T>* handler = new CallbackHandler_<T>(); // this calls the constructor with empty function
217  Callbacks::iterator callback = callbacks_.insert(std::make_pair(message_ID, boost::shared_ptr<CallbackHandler>(handler)));
218  callback_mutex_.unlock();
219 
220  // Wait for the message
221  if (handler->wait(timeout)) {
222  message = handler->get();
223  result = true;
224  }
225 
226  // Remove the callback handler
227  callback_mutex_.lock();
228  callbacks_.erase(callback);
229  callback_mutex_.unlock();
230  return result;
231  }
