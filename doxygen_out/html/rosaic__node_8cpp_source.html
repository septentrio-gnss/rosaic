397 void rosaic_node::ROSaicNode::InitializeIO()
398 {
399  ROS_DEBUG("Called InitializeIO() method");
400  boost::smatch match;
401  // In fact: smatch is a typedef of match_results<string::const_iterator>
402  if (boost::regex_match(device_, match, boost::regex("(tcp)://(.+):(\\d+)")))
403  // \d means decimal, however, in the regular expression, the \ is a special character, which needs to be escaped on its own as well..
404  // Note that regex_match can be used with a smatch object to store results, or without. In any case, true is returned if and only if it matches the !complete! string.
405  {
406  // The first sub_match (index 0) contained in a match_result always represents the full match within a target sequence made by a regex,
407  // and subsequent sub_matches represent sub-expression matches corresponding in sequence to the left parenthesis delimiting the sub-expression in the regex,
408  // i.e. $n Perl is equivalent to m[n] in boost regex.
409  std::string proto(match[1]);
410  if (proto == "tcp")
411  {
412  tcp_host_ = match[2];
413  tcp_port_ = match[3];
414 
415  serial_ = false;
416  boost::thread temporary_thread(boost::bind(&ROSaicNode::Connect, this));
417  temporary_thread.detach();
418  }
419  else
420  {
421  {
422  std::stringstream ss;
423  ss << "Protocol '" << proto << "' is unsupported.";
424  ROS_ERROR("%s", ss.str().c_str());
425  }
426  }
427  }
428  else
429  {
430  serial_ = true;
431  boost::thread temporary_thread(boost::bind(&ROSaicNode::Connect, this));
432  temporary_thread.detach();
433  }
434  ROS_DEBUG("Leaving InitializeIO() method");
435 }

452 void rosaic_node::ROSaicNode::Reconnect(const ros::TimerEvent& event)
453 {
454  ROS_DEBUG("Called Reconnect() method");
455  if (connected_ == true)
456  {
457  reconnect_timer_.stop();
458  ROS_DEBUG("Stopped ROS timer since successully connected.");
459  }
460  else
461  {
462  if (serial_)
463  {
464  bool initialize_serial_return = false;
465  try
466  {
467  ROS_INFO("Connecting serially to device %s, targeted baudrate: %u", device_.c_str(), baudrate_);
468  initialize_serial_return = IO.InitializeSerial(device_, baudrate_, hw_flow_control_);
469  }
470  catch (std::runtime_error& e)
471  {
472  {
473  std::stringstream ss;
474  ss << "IO.InitializeSerial() failed for device " << device_ << " due to: " << e.what();
475  ROS_ERROR("%s", ss.str().c_str());
476  }
477  }
478  if (initialize_serial_return)
479  {
480  boost::mutex::scoped_lock lock(connection_mutex_);
481  connected_ = true;
482  lock.unlock();
483  connection_condition_.notify_one();
484  }
485  }
486  else
487  {
488  bool initialize_tcp_return = false;
489  try
490  {
491  ROS_INFO("Connecting to tcp://%s:%s ...", tcp_host_.c_str(), tcp_port_.c_str());
492  initialize_tcp_return = IO.InitializeTCP(tcp_host_, tcp_port_);
493  }
494  catch (std::runtime_error& e)
495  {
496  {
497  std::stringstream ss;
498  ss << "IO.InitializeTCP() failed for host " << tcp_host_ << " on port " << tcp_port_ << " due to: " << e.what();
499  ROS_ERROR("%s", ss.str().c_str());
500  }
501  }
502  if (initialize_tcp_return)
503  {
504  boost::mutex::scoped_lock lock(connection_mutex_);
505  connected_ = true;
506  lock.unlock();
507  connection_condition_.notify_one();
508  }
509  }
510  }
511  ROS_DEBUG("Leaving Reconnect() method");
512 }
