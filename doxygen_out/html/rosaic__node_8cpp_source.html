72 void rosaic_node::ROSaicNode::InitializeIO()
73 {
74  ROS_DEBUG("Called InitializeIO() method");
75  boost::smatch match;
76  // In fact: smatch is a typedef of match_results<string::const_iterator>
77  if (boost::regex_match(device_, match, boost::regex("(tcp|udp)://(.+):(\\d+)")))
78  // \d means decimal, however, in the regular expression, the \ is a special character, which needs to be escaped on its own as well..
79  // Note that regex_match can be used with a smatch object to store results, or without. In any case, true is returned if and only if it matches the !complete! string.
80  {
81  // The first sub_match (index 0) contained in a match_result always represents the full match within a target sequence made by a regex,
82  // and subsequent sub_matches represent sub-expression matches corresponding in sequence to the left parenthesis delimiting the sub-expression in the regex,
83  // i.e. $n Perl is equivalent to m[n] in boost regex.
84  std::string proto(match[1]);
85  if (proto == "tcp")
86  {
87  std::string host(match[2]);
88  std::string port(match[3]);
89  ROS_INFO("Connecting to %s://%s:%s ...", proto.c_str(), host.c_str(), port.c_str());
90  try
91  {
92  IO.InitializeTCP(host, port);
93  }
94  catch (std::runtime_error& e)
95  {
96  throw std::runtime_error("IO.InitializeTCP() failed for host " + host + " on port " + port + " due to: " + e.what());
97  }
98  }
99  else
100  {
101  throw std::runtime_error("Protocol '" + proto + "' is unsupported");
102  }
103  }
104  else
105  {
106  // To be modified here, or clarified: how to use reconnect_delay_s properly? Is respawn (roslaunch parameter) enough?
107  //ROS_DEBUG("Setting timer for calling InitializeSerial() method");
108  //nh->param("reconnect_delay_s", reconnect_delay_s_, 0.5f);
109  //reconnect_timer_ = nh->createTimer(ros::Duration(reconnect_delay_s_), &ROSaicNode::Reconnect, this);
110  //reconnect_timer_.start();
111  //ROS_DEBUG("Started timer"); //not printed if error in callback of course
112  //ros::spin(); // otherwise callback will never be called, with ros::spin i cannot leave InitializeIO(), yet with ros::spinOnce can enter reconnect() even once
113  try
114  {
115  IO.InitializeSerial(device_, baudrate_);
116  }
117  catch (std::runtime_error& e)
118  {
119  throw std::runtime_error("IO.InitializeSerial() failed for device " + device_ + " due to: " + e.what());
120  }
121  }
122  ROS_DEBUG("Leaving InitializeIO()");
123 }
