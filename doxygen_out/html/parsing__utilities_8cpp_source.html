225  double UTCDoubleToSeconds(double utc_double)
226  {
227  uint32_t hours = static_cast<uint32_t>(utc_double) / 10000;
228  uint32_t minutes = (static_cast<uint32_t>(utc_double) - hours * 10000) / 100;
229  double seconds = utc_double - static_cast<double>(hours * 10000 + minutes * 100);
230  seconds += static_cast<double> (hours * 3600 + minutes * 60);
231  return seconds;
232  }

255  time_t UTCtoUnix(double utc_double)
256  {
257  time_t time_now = time(0);
258  struct tm * timeinfo;
259 
260  // The function localtime uses the value pointed by its argument to fill a tm structure with the values that represent the corresponding time, expressed for the local timezone.
261  timeinfo = gmtime(&time_now);
262 
263  uint32_t year = timeinfo->tm_year; // year, starting from 1900
264  uint32_t month = timeinfo->tm_mon; // months since January - [0,11]
265  uint32_t day = timeinfo->tm_mday; //day of the month - [1,31]
266  uint32_t hours = static_cast<uint32_t>(utc_double) / 10000;
267  uint32_t minutes = (static_cast<uint32_t>(utc_double) - hours * 10000) / 100;
268  uint32_t seconds = (static_cast<uint32_t>(utc_double) - hours * 10000 - minutes * 100);
269 
270  //ROS_DEBUG("Checking year %u, month %u, day %u", year, month, day);
271 
272  // Overwriting timeinfo with UTC time as extracted from utc_double..
273  timeinfo->tm_hour = hours; // hours since midnight - [0,23]
274  timeinfo->tm_min = minutes; // minutes after the hour - [0,59]
275  timeinfo->tm_sec = seconds; // seconds after the minute - [0,59]
276 
277  // Inverse of gmtime, the latter converts time_t (Unix time) to tm (UTC time)
278  time_t date = timegm(timeinfo);
279 
280  //ROS_DEBUG("Since 1970/01/01 %jd seconds have passed.\n", (intmax_t) date);
281  return date;
282  }

287  geometry_msgs::Quaternion ToQuaternion(double yaw, double pitch, double roll)
288  {
289  // Abbreviations for the angular functions
290  double cy = cos(yaw * 0.5);
291  double sy = sin(yaw * 0.5);
292  double cp = cos(pitch * 0.5);
293  double sp = sin(pitch * 0.5);
294  double cr = cos(roll * 0.5);
295  double sr = sin(roll * 0.5);
296 
297  geometry_msgs::Quaternion q;
298  q.w = cr * cp * cy + sr * sp * sy;
299  q.x = sr * cp * cy - cr * sp * sy;
300  q.y = cr * sp * cy + sr * cp * sy;
301  q.z = cr * cp * sy - sr * sp * cy;
302 
303  return q;
304  }
