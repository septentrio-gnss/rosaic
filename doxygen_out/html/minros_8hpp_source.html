212  void handle(mosaicMessage& mMessage)
213  {
214  // !!Find!! the callback handler for the message & decode it
215  boost::mutex::scoped_lock lock(callback_mutex_);
216  Callbacks::key_type key = mMessage.MessageID();
217  for (Callbacks::iterator callback = callbacks_.lower_bound(key);
218  callback != callbacks_.upper_bound(key); ++callback)
219  callback->second->handle(mMessage);
220  }
221 
222  template <typename T>
223  // There is a smalll overhead that comes with a boost::function<void(const T&)> (i.e. Callback) instead of another template that would make it a true functor.
224  // Passing raw functions as arguments to other functions is not recommended, needs e.g. std::transform etc.
225  bool poll(T& message, std::string message_ID, const boost::posix_time::time_duration& timeout) {
226  bool result = false;
227  // Create a callback handler for this message
228  callback_mutex_.lock();
229  CallbackHandler_<T>* handler = new CallbackHandler_<T>(); // this calls the constructor with empty function
230  Callbacks::iterator callback = callbacks_.insert(std::make_pair(message_ID, boost::shared_ptr<CallbackHandler>(handler)));
231  callback_mutex_.unlock();
232 
233  // Wait for the message
234  if (handler->wait(timeout)) {
235  message = handler->get();
236  result = true;
237  }
238 
239  // Remove the callback handler
240  callback_mutex_.lock();
241  callbacks_.erase(callback);
242  callback_mutex_.unlock();
243  return result;
244  }

246  void readCallback(uint8_t* data, std::size_t& size)
247  {
248  mosaicMessage mMessage(data, size);
249  // Read !all! (there might be many) the messages in buffer
250  while (mMessage.search() != mMessage.end() && mMessage.found()) {
251  if (debug >= 3) {
252  // Print the received bytes, on level debug, or just show messageID..
253  if (mMessage.pos()[1] == SEP_SYNC_BYTE_2)
254  {
255  ROS_DEBUG("Driver reading SBF block %s with %lu bytes...", mMessage.MessageID(), unsigned long (mMessage.block_length())); //recall: long is at least 32 bits
256  }
257  else
258  {
259  std::ostringstream oss;
260  boost::char_separator<char> sep(",");
261  typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
262  std::size_t end_point = std::min(static_cast<std::size_t>(mMessage.end() - mMessage.pos()), static_cast<std::size_t>(82));
263  std::string block_in_string(reinterpret_cast<char*>(mMessage.pos()), end_point); // Syntax: string (const char* s, size_t n); size_t is either 2 or 8 bytes, depending on your system
264  // NMEA 0183 messages are at most 82 characters long, , including the $ or ! starting character and the ending <LF>
265  tokenizer tokens(block_in_string, sep);
266  for (tokenizer::iterator it = tokens.begin();
267  *it != 0x0A; ++it)
268  oss << boost::format("%s") % *it << " ";
269  ROS_DEBUG("Driver reading NMEA message:\n%s", oss.str());
270  }
271  }
272 
273  handle(mMessage);
274  }
275 
276  // delete read bytes from input buffer
277  std::copy(mMessage.pos(), mMessage.end(), data); //Recall: The first and second parameter specify the range of elements to copy, while the third parameter specifies the beginning of the destination range.
278  size -= mMessage.pos() - data; // data was never modified
279  }
