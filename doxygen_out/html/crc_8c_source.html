42 uint16_t CRC_compute16CCITT (const void *buf, size_t buf_length) // the CRC we choose is 2 bytes, remember, hence uint16_t..
43 {
44  uint32_t i; // since i might become rather large, after all..
45  uint16_t crc = 0; // seed is 0, as suggested by ref guide. computed in the forward direction
46  const uint8_t *buf8 = (const uint8_t *) buf; // Convert the type to access by byte. RHS is pointer to constant type uint8_t, pointing to the address buf
47  // later in the code this address buf8 is used to write/read an actual uint8_t value to/from there.
48  // uint8_t x = address_at_eeprom_location[3]: This would read the 3rd uint8_t starting from address 10 (so at address 13, since addresses work byte-wise) in ram into the variable x
49  for (i=0; i<buf_length; i++) {
50  crc = (crc << 8) ^ CRCLookUp[ (crc >> 8) ^ buf8[i] ];
51  // The ^ (bitwise XOR) in C or C++ takes two numbers as operands and does XOR on every bit of two numbers. The result of XOR is 1 if the two bits are different.
52  // The << (left shift) in C or C++ takes two numbers, left shifts the bits of the first operand, the second operand decides the number of places to shift.
53  // The >> (right shift) in C or C++ takes two numbers, right shifts the bits of the first operand, the second operand decides the number of places to shift; you can just loose the smallest values if big-endian
54  // The left shift and right shift operators should not be used for negative numbers
55  // The left-shift and right-shift operators are equivalent to multiplication and division by 2 respectively, hence only rightshift is non-exact (remainder is not retained)
56  // CRC_SBF_LookUp contains is constructed from truncated polynomial (divisor)
57  // The above implements a kind of CRC 32 algorithm, efficient, fast
58  }
59 
60  return crc;
61 }
