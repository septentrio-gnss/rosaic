53 nmea_msgs::GpggaPtr rosaic_driver::GpggaParser::ParseASCII(const rosaic_driver::NMEASentence& sentence) noexcept(false)
54 {
55  //ROS_DEBUG("Just testing that first entry is indeed what we expect it to: %s", sentence.get_body()[0].c_str());
56  // Check the length first, which should be 15 elements (if station ID available, otherwise missing field, not empty field) or 14 elements (if station ID not avaiable).
57  const size_t MAX_LEN = 15;
58  const size_t MIN_LEN = 14;
59  if (sentence.get_body().size() > MAX_LEN || sentence.get_body().size() < MIN_LEN)
60  {
61  std::stringstream error;
62  error << "Expected GPGGA length " << MIN_LEN << " <= length <= "
63  << MAX_LEN << ", actual length = " << sentence.get_body().size();
64  throw ParseException(error.str());
65  }
66 
67  nmea_msgs::GpggaPtr msg = boost::make_shared<nmea_msgs::Gpgga>();
68  msg->header.frame_id = frame_id;
69 
70  msg->message_id = sentence.get_body()[0];
71 
72  if (sentence.get_body()[1].empty() || sentence.get_body()[1] == "0")
73  {
74  msg->utc_seconds = 0;
75  }
76  else
77  {
78  double utc_double;
79  if (string_utilities::ToDouble(sentence.get_body()[1], utc_double))
80  {
81  //ROS_DEBUG("utc_double is %f", (float) utc_double);
82  msg->utc_seconds = UTCDoubleToSeconds(utc_double);
83 
84  // The Header's Unix Epoch time stamp
85  time_t unix_time_seconds = UTCtoUnix(utc_double);
86  uint32_t unix_time_nanoseconds = (static_cast<uint32_t>(utc_double*100)%100)*10000; // Assumes there are two digits after the decimal point in NMEA UTC time
87  msg->header.stamp.sec = unix_time_seconds;
88  msg->header.stamp.nsec = unix_time_nanoseconds;
89  }
90  else
91  {
92  throw rosaic_driver::ParseException("Error parsing UTC seconds in GPGGA"); // E.g. if one of the time fields of the string is empty
93  }
94  }
95 
96  bool valid = true;
97 
98  double latitude = 0.0;
99  valid = valid && rosaic_driver::ParseDouble(sentence.get_body()[2], latitude);
100  msg->lat = ConvertDMSToDegrees(latitude);
101 
102  double longitude = 0.0;
103  valid = valid && rosaic_driver::ParseDouble(sentence.get_body()[4], longitude);
104  msg->lon = ConvertDMSToDegrees(longitude);
105 
106  msg->lat_dir = sentence.get_body()[3];
107  msg->lon_dir = sentence.get_body()[5];
108  valid = valid && rosaic_driver::ParseUInt32(sentence.get_body()[6], msg->gps_qual);
109  valid = valid && rosaic_driver::ParseUInt32(sentence.get_body()[7], msg->num_sats);
110  //ROS_DEBUG("Valid is %s so far with number of satellites in use being %s", valid ? "true" : "false", sentence.get_body()[7].c_str());
111 
112  valid = valid && rosaic_driver::ParseFloat(sentence.get_body()[8], msg->hdop);
113  valid = valid && rosaic_driver::ParseFloat(sentence.get_body()[9], msg->alt);
114  msg->altitude_units = sentence.get_body()[10];
115  valid = valid && rosaic_driver::ParseFloat(sentence.get_body()[11], msg->undulation);
116  msg->undulation_units = sentence.get_body()[12];
117  valid = valid && rosaic_driver::ParseUInt32(sentence.get_body()[13], msg->diff_age);
118  if (sentence.get_body().size() == MAX_LEN)
119  {
120  msg->station_id = sentence.get_body()[14];
121  }
122  else
123  {
124  msg->station_id = "";
125  }
126 
127  if (!valid)
128  {
129  was_last_gpgga_valid_ = false;
130  throw ParseException("GPGGA message was invalid.");
131  }
132 
133  // If we made it this far, we successfully parsed the message and will consider it to be valid.
134  was_last_gpgga_valid_ = true;
135 
136  return msg;
137 }
